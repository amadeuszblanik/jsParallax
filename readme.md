new JsParallax([
    {
        selector: ".box",
        values: [{
            name: "xyz",
            from: 0,
            to: 90,
            unit: "deg"
        }, {
            name: "x",
            from: 0,
            to: 90,
            unit: "deg"
        }]
    }
]).mount();
